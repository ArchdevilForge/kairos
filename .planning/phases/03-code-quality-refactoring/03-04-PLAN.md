    class ConfigHandler:
        """Handles config updates, symbol sync, exchange reload, auto-refresh."""

        def __init__(self, sentry: "PriceSentry"):

        def enqueue_config_update(self, event: "ConfigUpdateEvent") -> None:
            """Receive config updates from ConfigManager on background threads."""
            try:
                self._sentry._config_events.put_nowait(event)
            except Exception:
                self._sentry._config_events.put(event)

        def process_config_updates(self) -> None:
            """Drain pending config updates and apply each."""
            from queue import Empty
            while True:
                try:
                    event = self._sentry._config_events.get_nowait()
                except Empty:
                    break
                self.apply_config_update(event)

        def apply_config_update(self, event: "ConfigUpdateEvent") -> None:
            """Apply a single config update event."""
            start_time = time.time()
            changed_keys = ", ".join(sorted(event.diff.changed_keys)) or "<none>"
            logging.info("Processing configuration update; changed keys: %s", changed_keys)
            if event.warnings:
                for warning in event.warnings:
                    logging.warning("Configuration warning: %s", warning)
            with self._sentry._config_lock:
                self._sentry.config = event.new_config
            self.refresh_runtime_settings()
            if event.diff.requires_symbol_reload:
                self.reload_runtime_components(event)
            elapsed = time.time() - start_time
            if elapsed > 5:
                logging.warning("Configuration update processing exceeded 5s target: %.2fs", elapsed)
            else:
                logging.info("Configuration update applied in %.2fs", elapsed)

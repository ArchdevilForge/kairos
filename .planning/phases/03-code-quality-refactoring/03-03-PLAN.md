**R11 Optimization — asyncio.Event for anomaly events:**
Current: Main loop calls `_process_anomaly_events()` then sleeps 1s. If anomaly happens 0.1s after processing, waits 0.9s before next check.
Target: Use `asyncio.Event` — detector callback sets event when pushing to queue, main loop does `await asyncio.wait_for(event.wait(), timeout=1.0)` instead of `await asyncio.sleep(1)`. Latency drops from ~1s to <10ms.

Implementation: Add `self._anomaly_event = asyncio.Event()` to PriceSentry.__init__. In the detector callback `_on_anomaly`, after `put_nowait(event)`, call `self._anomaly_event.set()`. In main loop, replace `await asyncio.sleep(1)` with:
```python
try:
    await asyncio.wait_for(self._anomaly_event.wait(), timeout=1.0)
except asyncio.TimeoutError:
    pass
self._anomaly_event.clear()
```

    ```python
    """Main monitoring loop for PriceSentry — extracted from sentry.py."""
    from __future__ import annotations
    import asyncio
    import logging
    import time
    from typing import TYPE_CHECKING, Any

    from pwatch.core.alert_formatter import AlertFormatter
    from pwatch.utils.error_handler import ErrorSeverity, error_handler
    from pwatch.utils.monitor_top_movers import monitor_top_movers
    from pwatch.utils.cache_manager import notification_cooldown

    if TYPE_CHECKING:
        from pwatch.core.sentry import PriceSentry

    class MainLoopHandler:
        """Handles the main monitoring loop, delegated from PriceSentry.run()."""

        def __init__(self, sentry: "PriceSentry"):
            self._sentry = sentry
            self._warmup_seconds = 5
            self._ws_check_interval = 60  # seconds between WS health checks
            self._anomaly_event = asyncio.Event()  # R11: signals new anomaly events

        async def run(self) -> None:
            """Main monitoring loop — delegated from PriceSentry.run()."""
            # Start WebSocket
            self._sentry.exchange.start_websocket(self._sentry.matched_symbols)
            logging.info(f"Started WebSocket connection for {len(self._sentry.matched_symbols)} symbols")

            # Warmup: wait for initial price data
            logging.info(f"Waiting {self._warmup_seconds}s for WebSocket to collect initial price data...")
            await asyncio.sleep(self._warmup_seconds)

            last_check_time = time.time()
            last_ws_check_time = time.time()

            try:
                logging.info("Entering main loop, starting price movement monitoring")
                minutes_snapshot, _, check_interval, _, _ = self._sentry._snapshot_runtime_state()
                interval_minutes = max(check_interval / 60, 1)
                logging.info("Check interval set to %.2f minutes (%s seconds)", interval_minutes, int(check_interval))

                while True:
                    self._sentry._process_config_updates()
                    await self._process_anomaly_events()
                    if self._sentry._auto_mode:
                        self._sentry._check_auto_refresh()

                    (minutes_snapshot, threshold_snapshot, check_interval, symbols_snapshot, notification_filter_snapshot) = self._sentry._snapshot_runtime_state()
                    current_time = time.time()

                    # Price monitoring at check_interval
                    if current_time - last_check_time >= check_interval:
                        await self._check_price_movers(minutes_snapshot, threshold_snapshot, check_interval, symbols_snapshot, notification_filter_snapshot)
                        last_check_time = current_time

                    # WebSocket health check every 60s
                    if current_time - last_ws_check_time >= self._ws_check_interval:
                        last_ws_check_time = current_time
                        await self._check_websocket_health()

                    # R11: Wait for anomaly event or timeout (instead of sleep(1))
                    try:
                        await asyncio.wait_for(self._anomaly_event.wait(), timeout=1.0)
                    except asyncio.TimeoutError:
                        pass
                    self._anomaly_event.clear()

            except KeyboardInterrupt:
                logging.info("Received keyboard interrupt. Shutting down...")
            finally:
                try:
                    self._sentry.exchange.close()
                except Exception as e:
                    error_handler.handle_api_error(e, {"component": "PriceSentry", "operation": "cleanup"}, ErrorSeverity.WARNING)

        async def _check_price_movers(self, minutes, threshold, check_interval, symbols, notification_filter) -> None:
            """Check for top price movers via REST API."""
            logging.info("Checking price movements, %s seconds since last check", ...)
            if not symbols:
                logging.warning("No symbols available for monitoring")
                return
            try:
                result = await monitor_top_movers(minutes, symbols, threshold, self._sentry.exchange, self._sentry.config, notification_filter, cooldown_manager=notification_cooldown)
                if result:
                    grouped_events = AlertFormatter.group_batch_events(result)  # uses extracted formatter
                    logging.info("Detected %s batch movement event(s) exceeding threshold", len(result))
                    for symbol, sym_events in grouped_events.items():
                        message = AlertFormatter.format_combined_alert(symbol, sym_events)  # uses extracted formatter
                        if message:
                            send_result = self._sentry._send_alert(symbol, message)
                            if send_result["success"]:
                                logging.info("Batch movement alert sent successfully for %s", symbol)
                            else:
                                logging.warning("Batch movement alert failed for %s: %s", symbol, send_result["reason"])
                else:
                    logging.info("No price movements exceeding threshold detected")
            except Exception as e:
                error_handler.handle_api_error(e, {"component": "PriceSentry", "operation": "monitor_top_movers"}, ErrorSeverity.ERROR)
                logging.error(f"Error in price monitoring: {e}")

        async def _check_websocket_health(self) -> None:
            """Check WebSocket connection health, reconnect if needed."""
            logging.debug("Checking WebSocket connection status")
            if not self._sentry.exchange.ws_connected:
                self._sentry._ws_consecutive_failures += 1
                logging.warning(f"WebSocket disconnected (连续失败: {self._sentry._ws_consecutive_failures}/{self._sentry._ws_max_failures}), attempting reconnect")
                try:
                    self._sentry.exchange.check_ws_connection()
                    if self._sentry.exchange.ws_connected:
                        self._sentry._ws_consecutive_failures = 0
                        self._sentry._ws_alert_sent = False
                        logging.info("WebSocket reconnected successfully")
                except Exception as e:
                    error_handler.handle_network_error(e, {"component": "PriceSentry", "operation": "websocket_reconnect"}, ErrorSeverity.WARNING)
                if self._sentry._ws_consecutive_failures >= self._sentry._ws_max_failures and not self._sentry._ws_alert_sent:
                    alert_msg = f"⚠️ **WebSocket 连接异常**\n\n连续失败 {self._sentry._ws_consecutive_failures} 次，监控可能中断。请检查网络连接。"
                    try:
                        send_result = self._sentry.notifier.send(alert_msg)
                        if send_result["success"]:
                            self._sentry._ws_alert_sent = True
                            logging.warning("WebSocket failure alert sent: %s consecutive failures", self._sentry._ws_consecutive_failures)
                        else:
                            logging.error("Failed to send WS alert: %s", send_result["reason"])
                    except Exception as e:
                        logging.error(f"Failed to send WS alert: {e}")
            else:
                if self._sentry._ws_consecutive_failures > 0:
                    logging.info(f"WebSocket connection restored after {self._sentry._ws_consecutive_failures} failures")
                self._sentry._ws_consecutive_failures = 0
                self._sentry._ws_alert_sent = False

        async def _process_anomaly_events(self) -> None:
            """Drain the anomaly event queue, combine per-symbol, and send notifications."""
            # EXACT copy of _process_anomaly_events from sentry.py (lines 373-406)
            from queue import Empty
            events = []
            while True:
                try:
                    events.append(self._sentry._anomaly_events.get_nowait())
                except Empty:
                    break
            if not events:
                return
            severity_rank = {"HIGH": 3, "MEDIUM": 2, "LOW": 1}
            by_symbol: dict[str, dict[str, Any]] = {}
            for ev in events:
                sym_events = by_symbol.setdefault(ev.symbol, {})
                existing = sym_events.get(ev.event_type)
                if existing is None or severity_rank.get(ev.severity, 0) > severity_rank.get(existing.severity, 0):
                    sym_events[ev.event_type] = ev
            for symbol, sym_events in by_symbol.items():
                if not notification_cooldown.should_notify(symbol):
                    logging.info("Skipping realtime alert for %s; notification cooldown active", symbol)
                    continue
                has_price_confirmation = bool(sym_events.get("price_velocity") or sym_events.get("batch_move"))
                if sym_events.get("volume_spike") and not has_price_confirmation:
                    continue
                message = AlertFormatter.format_combined_alert(symbol, sym_events)
                if message:
                    send_result = self._sentry._send_alert(symbol, message)
                    if not send_result["success"]:
                        logging.warning("Realtime alert failed for %s: %s", symbol, send_result["reason"])

        def signal_anomaly_event(self) -> None:
            """R11: Called by detector callback to signal new anomaly event."""
            self._anomaly_event.set()
    ```

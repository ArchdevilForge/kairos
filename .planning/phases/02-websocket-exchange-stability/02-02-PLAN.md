    ```python
    def _reload_runtime_components(self, event: ConfigUpdateEvent) -> None:
        exchange_name = self.config.get("exchange", "binance")
        logging.info(
            "Reloading exchange and symbol set due to config update (keys: %s)",
            ", ".join(sorted(event.diff.changed_keys)) or "<none>",
        )

        # Step 1: Prepare new exchange OFF the main reference
        try:
            new_exchange = get_exchange(exchange_name)
        except Exception as exc:
            error_handler.handle_config_error(
                exc,
                {"component": "PriceSentry", "operation": "exchange_reload", "exchange": exchange_name},
                ErrorSeverity.CRITICAL,
            )
            logging.error("Exchange reload aborted: %s", exc)
            return

        # Step 2: Register detectors on new exchange BEFORE making it live
        try:
            self._register_detectors_on(new_exchange)
        except Exception as exc:
            logging.error("Failed to register detectors on new exchange: %s", exc)
            new_exchange.close()
            return

        # Step 3: Sync symbols on new exchange (uses new_exchange, not self.exchange)
        try:
            new_symbols = self._sync_symbols_for_exchange(new_exchange, exchange_name)
        except ValueError as exc:
            error_handler.handle_config_error(
                exc,
                {"component": "PriceSentry", "operation": "symbol_reload", "exchange": exchange_name},
                ErrorSeverity.ERROR,
            )
            logging.error("Symbol reload aborted: %s", exc)
            new_exchange.close()
            return

        if not new_symbols:
            logging.warning("Symbol reload produced empty set; skipping exchange swap")
            new_exchange.close()
            return

        # Step 4: Atomic swap under lock
        with self._exchange_lock:
            old_exchange = self.exchange
            self.exchange = new_exchange
            self.matched_symbols = new_symbols
            self._rebuild_notification_filter_locked()

        # Step 5: Start WebSocket on new exchange (now live)
        try:
            self.exchange.start_websocket(new_symbols)
        except Exception as exc:
            error_handler.handle_network_error(
                exc,
                {"component": "PriceSentry", "operation": "websocket_restart", "symbols_count": len(new_symbols)},
                ErrorSeverity.ERROR,
            )
            logging.error("Failed to restart websocket after config change: %s", exc)
            # Rollback: restore old exchange
            with self._exchange_lock:
                self.exchange = old_exchange
            return

        # Step 6: Close old exchange after successful swap
        try:
            old_exchange.close()
        except Exception as exc:
            logging.warning("Failed to close old exchange cleanly after swap: %s", exc)

        logging.info("Exchange and symbol sets reloaded successfully (%s symbols)", len(new_symbols))
    ```

    ```python
    def _sync_symbols_for_exchange(self, exchange: "BaseExchange", exchange_name: str) -> List[str]:
        """Sync symbols for a given exchange instance. Returns symbol list without mutating self."""
        selection = self.config.get("notificationSymbols")
        is_auto = selection == "auto" or (not selection)

        if is_auto:
            auto_limit = self.config.get("autoModeLimit", 50)
            symbols = fetch_top_volume_symbols(
                exchange_name,
                limit=auto_limit,
                filters=self._auto_mode_filters(),
            )
            if not symbols:
                raise ValueError("Failed to fetch top volume symbols")
            return symbols

        # Manual mode
        available_symbols = load_usdt_contracts(exchange_name)
        if not available_symbols:
            return []

        if not isinstance(selection, list) or not selection:
            raise ValueError("Configuration must include at least one notification symbol or set to 'auto'")

        available_set = set(available_symbols)
        monitored: List[str] = []
        for raw_symbol in selection:
            if isinstance(raw_symbol, str) and raw_symbol.strip() in available_set:
                monitored.append(raw_symbol.strip())

        if not monitored:
            raise ValueError("No valid notification symbols remain after filtering")
        return monitored
    ```

    - Swap is atomic under `_exchange_lock`
    - Old exchange closed AFTER swap, so main loop never sees broken reference
    - Rollback on WS start failure restores old exchange
    - No silent returns that leave state inconsistent

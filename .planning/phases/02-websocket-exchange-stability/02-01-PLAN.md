       ```python
       def _stop_websocket_thread(self) -> None:
           """Stop only the WebSocket thread, preparing for reconnection."""
           if self.ws_thread and self.ws_thread.is_alive():
               self._ws_stop_event.set()
               self.ws_thread.join(timeout=5)
               if self.ws_thread.is_alive():

           self._ws_stop_event.clear()

       def _stop_websocket_thread(self) -> None:
           """Stop only the WebSocket thread, preparing for reconnection."""
           if self.ws_thread and self.ws_thread.is_alive():
               # Signal the WebSocket loop to exit
               self.ws_connected = False  # trigger exit in _ws_connect
               self.ws_thread.join(timeout=5)
               if self.ws_thread.is_alive():
                   logging.warning(
                       "WebSocket thread did not exit within 5s timeout for %s",
                       self.exchange_name,
                   )
           self.ws_thread = None
       ```

       - Then: `self.start_websocket(symbols)` (this sets `self.running = True` internally)
